PASOS PARA CONFIGURAR DB SAKILA EN ORACLE

1-LOGGEARSE EN SQLPLUS COMO SYSDBA
	SQLPLUS / AS SYSDBA
2-COLOCAR EL CONTAINIER XEPDB1
	ALTER SESSION SET CONTAINER = XEPDB1;
3-CREAR TABLESPACE TBS_SAKILA EN CONTAINER
	CREATE TABLESPACE TBS_SAKILA datafile 'C:\app\Zito\product\18.0.0\oradata\XE\XEPDB1\TBS_SAKILA.DBF' size 20m autoextend on next 5m maxsize 100m online;
4-CREAR USUARIO
	CREATE USER usr_sakila identified by oracle default tablespace TBS_SAKILA quota unlimited on tbs_sakila; --PUEDE CAMBIAR EL USUARIO Y CLAVE
5-CREAR ROL
	CREATE ROLE DEVELOPER; --PUEDE CAMBIAR EL NOMBRE
6-ASIGNAR PRIVILEGIOS AL ROL
	GRANT CREATE SESSION, CREATE TABLE, ALTER ANY TABLE, DROP ANY TABLE, INSERT ANY TABLE, UPDATE ANY TABLE, DELETE ANY TABLE, SELECT ANY TABLE TO DEVELOPER;
7-ASIGNAR EL ROL CON LOS PRIVILEGIOS AL USUARIO CREADO (usr_sakila)
	GRANT DEVELOPER TO USR_SAKILA;
8-ABRIR SQLDEVELOPER 
	CREAR CONEXION CON LOS DATOS CORRESPONDIENTES
	ABRIR ARCHIVO DE SCHEMA Y EJECUTAR
	ABRIR ARCHIVO DE INSERT Y EJECUTAR
